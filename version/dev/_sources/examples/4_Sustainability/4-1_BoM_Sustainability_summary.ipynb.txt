   "source": [
    "# Define the criterion\n",
    "criterion = (transport_df_full[\"EE%\"] < 5.0) | (transport_df_full[\"CC%\"] < 5.0)\n",
    "\n",
    "# Aggregate all rows that meet the criterion\n",
    "transport_df_below_5_pct = transport_df_full.loc[criterion].sum(numeric_only=True).to_frame().T\n",
    "transport_df_below_5_pct[\"Name\"] = \"Other\"\n",
    "\n",
    "# Sort all rows that do not meet the criterion by embodied energy\n",
    "transport_df_over_5_pct = transport_df_full.loc[~(criterion)].sort_values(by=\"EE%\", ascending=False)\n",
    "\n",
    "# Concatenate the rows together\n",
    "transport_df = pd.concat([transport_df_over_5_pct, transport_df_below_5_pct], ignore_index=True)\n",
